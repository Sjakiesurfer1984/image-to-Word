import { useState } from "react";
import { FileUpload } from "@/components/FileUpload";
import { PhotoGrid } from "@/components/PhotoGrid";
import { Trash2 } from "lucide-react";

interface Photo {
  id: string;
  file: File;
  preview: string;
}

const Index = () => {
  const [photos, setPhotos] = useState<Photo[]>([]);
  const [discardedPhotos, setDiscardedPhotos] = useState<Photo[]>([]);

  const handleFilesSelected = (files: File[]) => {
    const newPhotos: Photo[] = files.map((file) => ({
      id: crypto.randomUUID(),
      file,
      preview: URL.createObjectURL(file),
    }));
    setPhotos((prev) => [...prev, ...newPhotos]);
  };

  const handlePhotosDiscarded = (photoIds: string[]) => {
    const photosToDiscard = photos.filter(photo => photoIds.includes(photo.id));
    setDiscardedPhotos((prev) => [...prev, ...photosToDiscard]);
    setPhotos((prev) => prev.filter(photo => !photoIds.includes(photo.id)));
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-background via-background to-muted/20">
      <div className="container mx-auto px-4 py-8 max-w-7xl">
        <header className="mb-8 text-center space-y-2">
          <div className="flex items-center justify-center space-x-2 mb-2">
            <Trash2 className="w-8 h-8 text-primary" />
            <h1 className="text-4xl font-bold bg-gradient-to-r from-primary to-accent bg-clip-text text-transparent">
              Photo Manager
            </h1>
          </div>
          <p className="text-lg text-muted-foreground">
            Select photos to discard and generate a Word document
          </p>
        </header>

        <div className="space-y-8">
          {photos.length === 0 ? (
            <FileUpload onFilesSelected={handleFilesSelected} />
          ) : (
            <>
              <div className="flex justify-between items-center">
                <p className="text-sm text-muted-foreground">
                  {photos.length} photo{photos.length > 1 ? "s" : ""} loaded
                </p>
                <FileUpload onFilesSelected={handleFilesSelected} />
              </div>
              <PhotoGrid
                photos={photos}
                discardedPhotos={discardedPhotos}
                onPhotosDiscarded={handlePhotosDiscarded}
              />
            </>
          )}
        </div>
      </div>
    </div>
  );
};

export default Index;
